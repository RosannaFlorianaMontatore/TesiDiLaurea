 package org.apache.commons.numbers.core;
 
 /**
  * Addition.
  *
  * @param <T> Type of elements.
  */
 public interface Addition<T> {
     /**
      * Binary addition.
      *
      * @param a Element.
      * @return {@code this + a}.
      */
     T add(T a);
 
     /**
      * Identity element.
      *
      * @return the field element such that for all {@code a},
      * {@code zero().add(a).equals(a)} is {@code true}.
      */
     T zero();
 
     /**
      * Additive inverse.
      *
      * @return {@code -this}.
      */
     T negate();
 }
