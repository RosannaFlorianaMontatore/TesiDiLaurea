 package org.apache.commons.statistics.inference;
 
 /**
  * Represents a non-equal comparison between two numbers.
  *
  * @see <a href="https://en.wikipedia.org/wiki/Inequality_(mathematics)">Inequality (Wikipedia)</a>
  * @since 1.1
  */
 public enum Inequality {
     /**
      * Represents a strict inequality.
      * <ul>
      * <li>{@code a < b}
      * <li>{@code a > b}
      * </ul>
      */
     STRICT,
 
     /**
      * Represents a non-strict inequality (numbers may be equal).
      * <ul>
      * <li>{@code a <= b}
      * <li>{@code a >= b}
      * </ul>
      */
     NON_STRICT
 }
